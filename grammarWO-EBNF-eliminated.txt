PROGRAM ::= "begin" DECLS "|" STATEMENT "end"
DECLS ::= "declare" DECLS'
DECLS' ::= ID "," DECLS'
DECLS' ::= epsilon
STATEMENT ::= STATEMENT' ";" STATEMENT
STATEMENT :: = epsilon
STATEMENT' ::= ID ":=" EXP

LEXP ::= ID 
LEXP ::= NAT 
LEXP ::= "(" EXP ")" 
LEXP ::= "-" EXP

EXP ::= LEXP PEXP
PEXP ::= "+" EXP PEXP 
PEXP ::= epsilon

EXP ::= LEXP MEXP
MEXP ::= "*" EXP MEXP
MEXP ::= epsilon

ID ::= [a-z][a-z0-9]*

NAT ::= [0] | [1-9][0-9]*

LAYOUT ::= [\ \n]*
